```rust
fn logsumexp(x: &Array1<f64>) -> f64 {
    let max_val = x.iter().cloned().fold(f64::NEG_INFINITY, f64::max);
    if max_val == f64::NEG_INFINITY { return f64::NEG_INFINITY; }
    let sum_exp: f64 = x.iter().map(|&v| (v - max_val).exp()).sum();
    max_val + sum_exp.ln()
}
```
