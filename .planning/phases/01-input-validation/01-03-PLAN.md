    1. Add to `ferroml-python/src/array_utils.rs`:

       ```rust
       /// Check that a 2D NumPy array contains no NaN or Inf values.
       /// Raises ValueError (not RuntimeError) consistent with sklearn conventions.
       pub fn check_array_finite(x: &PyReadonlyArray2<f64>) -> PyResult<()> {
           let arr = x.as_array();
           for ((row, col), val) in arr.indexed_iter() {
               if !val.is_finite() {
                   let kind = if val.is_nan() { "NaN" } else { "Inf" };
                   return Err(PyErr::new::<pyo3::exceptions::PyValueError, _>(
                       format!("Input array contains {} at position ({}, {})", kind, row, col)
                   ));
               }
           }
           Ok(())
       }

       /// Check that a 1D NumPy array contains no NaN or Inf values.
       pub fn check_array1_finite(x: &PyReadonlyArray1<f64>) -> PyResult<()> {
           let arr = x.as_array();
           for (idx, val) in arr.iter().enumerate() {
               if !val.is_finite() {
                   let kind = if val.is_nan() { "NaN" } else { "Inf" };
                   return Err(PyErr::new::<pyo3::exceptions::PyValueError, _>(
                       format!("Input array contains {} at position {}", kind, idx)
                   ));
               }
           }
           Ok(())
       }
       ```
       NOTE: Use `indexed_iter()` only if it is available on ndarray ArrayView. If not, use a manual row/col counter or `.iter().enumerate()` with manual position calculation.
