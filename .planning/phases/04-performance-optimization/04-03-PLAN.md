fn bench_ols_solver_paths(c: &mut Criterion) {
    let mut group = c.benchmark_group("OLS_Solver_Paths");
    // Tall dataset: n >> 2d -> Cholesky path
    for (n, p, label) in [
        (5000, 50, "tall_5000x50_cholesky"),
        (10000, 100, "tall_10000x100_cholesky"),
        // Wide dataset: n <= 2d -> QR path
        (200, 100, "wide_200x100_qr"),
        (100, 50, "sq_100x50_qr"),
    ] {
        let (x, y) = make_regression(n, p, Some(0.1), Some(42));
        group.bench_with_input(
            BenchmarkId::new("fit", label),
            &(x.clone(), y.clone()),
            |b, (x, y)| {
                b.iter(|| {
                    let mut ols = LinearRegression::new();
                    ols.fit(black_box(x), black_box(y)).unwrap()
                });
            },
        );
    }
    group.finish();
}
